    i = 0;
    while (i < size - 1)
    {
        j = 0;

Core Algorithm:

int is_valid(int *board, int row, int col)
{
    int i = 0;
    while (i < col)
    {
        if (board[i] == row)  /* Same row */
            return (0);
        if (ft_abs(board[i] - row) == ft_abs(i - col))  /* Diagonal */
            return (0);
        i++;
    }
    return (1);
}

int solve_helper(int *board, int col)
{
    int total = 0, row = 0;
    if (col >= 10) { /* Print solution */ return (1); }
    while (row < 10)
    {
        if (is_valid(board, row, col))
        {
            board[col] = row;
            total += solve_helper(board, col + 1);
            board[col] = -1;  /* Backtrack */
        }
        row++;
    }
    return (total);
}
